```python
async def post_asiento(session, fecha, concepto, lineas, metadata=None):
    async with session.begin():
        asiento = Asiento(fecha=fecha, concepto=concepto, metadata=metadata)
        session.add(asiento)
        await session.flush()  # get asiento.id
        for l in lineas:
            session.add(AsientoLinea(asiento_id=asiento.id, cuenta_id=l["cuenta_id"],
                                     debe=l["debe"], haber=l["haber"]))
        await session.flush()
        # Balance validation in Python (D-05 forbids SQL triggers)
        total_debe = sum(l["debe"] for l in lineas)
        total_haber = sum(l["haber"] for l in lineas)
        if total_debe != total_haber:
            raise ValueError(f"Asiento no cuadra: debe={total_debe} haber={total_haber}")
        # D-05: also validate single-moneda — all lineas reference cuentas with same moneda
        return asiento
```

### `apps/tours/api/app/services/commission.py` — Precedence resolver + `/simular`

**No analog.** Use **RESEARCH.md Pattern 3** (lines 423-479) verbatim:

```python
async def resolve_comision(session, vendedor_id, tour_id) -> float:
    # Precedence 1: vendedor+tour
    stmt = select(ComisionRegla).where(
        ComisionRegla.vendedor_id == vendedor_id,
        ComisionRegla.tour_id == tour_id)
    if regla := (await session.execute(stmt)).scalar_one_or_none():
        return regla.porcentaje
    # Precedence 2: vendedor only
    stmt = select(ComisionRegla).where(
        ComisionRegla.vendedor_id == vendedor_id,
        ComisionRegla.tour_id.is_(None))  # ← Pitfall 5: use .is_(None), not ==
    if regla := (await session.execute(stmt)).scalar_one_or_none():
        return regla.porcentaje
    # Precedence 3: tour only
    stmt = select(ComisionRegla).where(
        ComisionRegla.vendedor_id.is_(None),
        ComisionRegla.tour_id == tour_id)
    if regla := (await session.execute(stmt)).scalar_one_or_none():
        return regla.porcentaje
    # Precedence 4: default global (non-deletable, seeded on init)
    stmt = select(ComisionRegla).where(
        ComisionRegla.vendedor_id.is_(None),
        ComisionRegla.tour_id.is_(None))
    return (await session.execute(stmt)).scalar_one().porcentaje

async def simular_comision(session, vendedor_id, tour_id, monto) -> dict:
    porcentaje = await resolve_comision(session, vendedor_id, tour_id)
    return {"vendedor_id": vendedor_id, "tour_id": tour_id, "monto": monto,
            "porcentaje": porcentaje, "comision": monto * (porcentaje / 100)}
```

**Apply to**: POST `/ventas` (live preview calls `/simular`), POST `/liquidaciones/{id}/close` (D-09 margen × porcentaje).

### `apps/tours/api/app/dependencies.py` — `get_session` + `get_current_user` (JWT verify)

**No analog.** Use **RESEARCH.md** lines 793-813 pattern (FastAPI JWT validation):

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import jwt
from sqlalchemy.ext.asyncio import AsyncSession
from contextvars import ContextVar

from app.config import settings
from app.audit import current_user_id  # ContextVar — set here, read in before_flush

security = HTTPBearer()

async def get_current_user(credentials: HTTPAuthorizationCredentials = Depends(security)) -> dict:
    try:
        payload = jwt.decode(credentials.credentials,
                             settings.NEXTAUTH_SECRET, algorithms=["HS256"])
    except jwt.PyJWTError:
        raise HTTPException(status.HTTP_401_UNAUTHORIZED, "Token inválido")
    user = {"id": payload["sub"], "name": payload.get("name"),
            "role": payload.get("role")}
    current_user_id.set(int(user["id"]))  # feeds audit_log event listener (D-23)
    return user

def require_role(*roles):
    async def _check(user: dict = Depends(get_current_user)):
        if user["role"] not in roles:
            raise HTTPException(status.HTTP_403_FORBIDDEN, "No tienes permiso")
        return user
    return _check

async def get_session() -> AsyncSession:
    async with async_session_maker() as s:
        yield s
```

**Apply**: every router. `require_role("admin")` for `/audit-log` (D-24), `/usuarios`. `require_role("admin", "contabilidad")` for close/reopen liquidación. `get_current_user` everywhere else.

```typescript
import { getServerSession } from "next-auth";
import { authOptions } from "./auth";

export async function apiFetch(path: string, init: RequestInit = {}) {
  const session = await getServerSession(authOptions);
  const token = (session as any)?.user?.token;  // raw JWT from NextAuth JWT callback
  const res = await fetch(`http://tours-api:8000${path}`, {
    ...init,
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${token}`,
      ...(init.headers ?? {}),
    },
  });
  if (!res.ok) throw new Error(`API ${res.status}: ${await res.text()}`);
  return res.json();
}
```
